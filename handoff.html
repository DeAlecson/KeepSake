05 Data Models

-- products (Product 1 — premade drops)
CREATE TABLE products (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name        TEXT NOT NULL,
  slug        TEXT UNIQUE NOT NULL,
  type        TEXT CHECK (type IN ('necklace', 'keychain')),
  drop_num    INT NOT NULL,
  price_sgd   NUMERIC(8,2) NOT NULL,
  was_sgd     NUMERIC(8,2),
  colors      TEXT[] NOT NULL,      -- hex array [c1, c2, c3]
  sold_out    BOOLEAN DEFAULT false,
  created_at  TIMESTAMPTZ DEFAULT now()
);

-- waitlist
CREATE TABLE waitlist (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  email       TEXT UNIQUE NOT NULL,
  created_at  TIMESTAMPTZ DEFAULT now(),
  order_link  TEXT,                  -- unique link sent via email
  link_sent   BOOLEAN DEFAULT false
);

-- bespoke_orders (Product 2)
CREATE TABLE bespoke_orders (
  id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  user_email      TEXT NOT NULL,
  template        TEXT NOT NULL,     -- filled|heart|swirl|bundle|line|circle
  addons          TEXT[],            -- ['gold','flowers'] etc.
  attachment_key  TEXT NOT NULL,     -- stainless|string|silver-pl|rose-gold|gold-pl|keychain
  attachment_price NUMERIC(6,2) DEFAULT 0,
  total_sgd       NUMERIC(8,2) NOT NULL,
  status          TEXT DEFAULT 'pending_hair',
  -- pending_hair | hair_received | crafting | qa | shipped | delivered
  stripe_payment_id TEXT,
  nfc_chip_id     TEXT UNIQUE,       -- assigned after crafting
  tracking_number TEXT,
  created_at      TIMESTAMPTZ DEFAULT now(),
  updated_at      TIMESTAMPTZ DEFAULT now()
);

-- pets (Feature X)
CREATE TABLE pets (
  id          UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  order_id    UUID REFERENCES bespoke_orders(id),
  owner_id    UUID REFERENCES auth.users(id),
  nfc_chip_id TEXT UNIQUE NOT NULL,  -- links to bespoke_orders.nfc_chip_id
  name        TEXT NOT NULL,
  breed       TEXT NOT NULL,
  photo_url   TEXT,
  description TEXT,
  thought     TEXT,                  -- "small message from pet"
  social_url  TEXT,
  is_public   BOOLEAN DEFAULT true,
  created_at  TIMESTAMPTZ DEFAULT now(),
  updated_at  TIMESTAMPTZ DEFAULT now()
);

-- nfc_chips (device management)
CREATE TABLE nfc_chips (
  chip_id     TEXT PRIMARY KEY,      -- physical chip UID
  pet_id      UUID REFERENCES pets(id),
  order_id    UUID REFERENCES bespoke_orders(id),
  activated   BOOLEAN DEFAULT false,
  activated_at TIMESTAMPTZ
);

08 Feature X — NFC / QR System

NFC Chip Lifecycle
  1. During production, each bespoke keepsake is assigned a unique chip_id (NFC UID) programmed onto an NTAG215 or similar chip embedded in the resin.
  2. The chip is pre-programmed to redirect to https://keepsake.sg/nfc/{chip_id}.
  3. A QR code with the same destination is printed and placed in the box. The QR also encodes an owner secret: /nfc/{chip_id}?secret={ownerToken}.
  4. Registration flow: user scans QR (gets owner token) → taps NFC → POST /api/nfc/verify with both chip_id and ownerToken → backend confirms match → returns a session JWT for pet creation.
  5. Once registered, visiting /nfc/{chip_id} (without secret) redirects to the pet's public profile page /pets/{petId}.
  6. Only the original QR secret holder can update the profile (PATCH /api/pets/{id} requires the owner JWT).
// NFC redirect route: /nfc/[chipId]/page.tsx
export async function GET(req, { params }) {
  const { chipId } = params;
  const secret = new URL(req.url).searchParams.get('secret');

  const chip = await db.nfc_chips.findUnique({ where: { chip_id: chipId } });
  if (!chip) return notFound();

  // If secret present → registration flow
  if (secret && chip.owner_secret === secret) {
    return redirect(`/pets/register?chip=${chipId}&token=${secret}`);
  }

  // Public tap → pet profile
  if (chip.pet_id) return redirect(`/pets/${chip.pet_id}`);

  // Unregistered chip
  return redirect('/pets/register?chip=' + chipId);
}

09 Payments Integration

// Bespoke order payment flow
// 1. Client calls POST /api/orders/bespoke → creates order + PaymentIntent
// 2. Client uses Stripe Elements to collect card / Apple Pay / Google Pay
// 3. On payment success, Stripe webhook fires → update order.status = 'pending_hair'
// 4. Send email to customer with studio mailing address + order reference
// 5. Admin panel (future) marks hair_received → crafting → qa → shipped

// Pricing logic (server-side — never trust client)
const BASE_PRICE = 150_00; // cents SGD
const ADDON_PRICES = { gold: 30_00, silver: 10_00, flowers: 15_00 };
const ATTACHMENT_PRICES = {
  stainless: 0, string: 0,
  'silver-pl': 20_00, 'rose-gold': 30_00, 'gold-pl': 30_00,
  keychain: 0
};

function calcTotal(addons, attachment) {
  return BASE_PRICE
    + addons.reduce((s, a) => s + (ADDON_PRICES[a] ?? 0), 0)
    + (ATTACHMENT_PRICES[attachment] ?? 0);
}
